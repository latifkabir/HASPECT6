192 void THSLongPS::Analyse(){
193  //Construct the CM boosts
194  TVector3 CMboost=-fCM.BoostVector();
195  TLorentzRotation CMRot(CMboost);
196 
197  //loop over particles and check if they are going forward in CM
198  vector< Bool_t > whichForward(fSize); //whichForward[2]=kTRUE means particle 2 is going forward
199  fPtot=0;//total longitudinal momentum
200  for(Int_t ip=0;ip<fSize;ip++){
201  fP4s[ip]=(CMRot*(fP4s[ip])); //cm boost
202  whichForward[ip]=fP4s[ip].Z()>0;
203  fPtot+=fP4s[ip].Z()*fP4s[ip].Z();
204  }
205  fPtot=TMath::Sqrt(fPtot);
206 
207  for(fSector=0;fSector<fNSector;fSector++)
208  if(fIsForward[fSector]==whichForward) break;
209

215 
216  TLorentzVector pBot;
217  for(UInt_t ip=0;ip<fIBot[fSector].size();ip++) pBot+=fP4s[fIBot[fSector][ip]];
