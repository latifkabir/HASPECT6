420 TList* THSOutput::CopyDirtoList(TDirectory *source) {
421  //copy all objects and subdirs of directory source as a tlist
422  //This should be easy, but for some reason is not.
423  //This code is based on tutorials/io/copyFiles.C
424  TList* list=new TList();
425  list->SetName(source->GetName());
426  list->SetOwner();
427  TDirectory *savdir = gDirectory;
428  //loop on all entries of this directory
429  TKey *key=0;
430  TIter nextkey(source->GetListOfKeys());
431  while ((key = (TKey*)nextkey())) {
432  const char *classname = key->GetClassName();
433  TClass *cl = gROOT->GetClass(classname);
434  if (!cl) continue;
435  if (cl->InheritsFrom(TDirectory::Class())) {
436  source->cd(key->GetName());
437  TDirectory *subdir = gDirectory;
438  list->Add(CopyDirtoList(subdir));
439  } else if (cl->InheritsFrom(TTree::Class())) {
440  TTree *T = (TTree*)source->Get(key->GetName());
441  TTree *newT = T->CloneTree(-1,"fast");
442  list->Add(newT);
443  } else {
444  source->cd();
445  TObject *obj = key->ReadObj();
446  list->Add(obj);
447  }
448  }
449  savdir->cd();
450  return list;
