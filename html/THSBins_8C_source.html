211 Int_t THSBins::FindBin(TVectorD vals){
212  //Loop over each axis and find bin for each
213  Bool_t InLimits=kTRUE;
214  for(Int_t iA=0;iA<fNaxis;iA++)//first check var is within variable ranges
215  if(vals[iA]<fVarAxis[iA].GetXmin()||vals[iA]>fVarAxis[iA].GetXmax()) InLimits=kFALSE;
216  if(!InLimits) {return -1;}
217  //now find bin for each axis
218  vector<Int_t> vBin(fNaxis); //store for the bin number of each axis
219  for(Int_t iA=0;iA<fNaxis;iA++){//loop over vars/axis
220  vBin[iA]=1+TMath::BinarySearch(fVarAxis[iA].GetNbins(),fVarAxis[iA].GetXbins()->GetArray(),vals[iA]);
221  }
222  //now have the bin for each axis, find single bin
223  Int_t theBin=-1;
224  for(Int_t iA1=0;iA1<fNaxis-1;iA1++){
225  Int_t tbin=vBin[iA1]-1;//-1 as bin indexing starts at 1 with 0 underflow
226  for(Int_t iA2=iA1+1;iA2<fNaxis;iA2++)
227  tbin*=fVarAxis[iA2].GetNbins();
228  theBin+=tbin;
229  }
230  theBin+=vBin[fNaxis-1];
231  return theBin;
232 }
