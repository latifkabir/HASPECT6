187  //create 1D template of x variable

190  Double_t weight=1;

195  //Fill first y bin of 2D hist (no smearing)

197  fRHist->Fill(fRHist->GetXaxis()->GetBinCenter(jtemp),fRHist->GetYaxis()->GetBinCenter(1),his1->GetBinContent(jtemp));
198 
199  //Loop over bins of smearing parameter
200  TF1 gausnX("gausnX","gausn(0)",fRHist->GetXaxis()->GetXmin(),fRHist->GetXaxis()->GetXmax());
201  for(Int_t ia=2;ia<=fRHist->GetNbinsY();ia++){//alpha bins
202  Double_t vAlphb=fRHist->GetYaxis()->GetBinCenter(ia);
203  Double_t vAlph=fRHist->GetYaxis()->GetBinLowEdge(ia);
204  for(Int_t itemp=1;itemp<=fRHist->GetNbinsX();itemp++){//fill with Gaussian function
205  Double_t vari=fRHist->GetXaxis()->GetBinCenter(itemp);
206  Double_t NX=his1->GetBinContent(itemp);
207  if(!NX) continue;
208  gausnX.SetParameters(NX,vari,vAlphb);
209  //Add gaussian for each non xero bin with width = yaxis value
210  for(Int_t jtemp=1;jtemp<=fRHist->GetNbinsX();jtemp++){//fill with Gaussian function

212  fRHist->Fill(varj,vAlphb,gausnX.Eval(varj));
213  }
214  }
215  //Cannot calculate covaraince if pdf==0 for some events
