145 inline void THSKinematics::PolPhotoCMDecay(){
146  // From Daria Sokhan thesis
147  // Assuming polarisation vector is (0,1,0) (the E vector) and B vector (1,0,0), these are components of the boosted E-vector.
148 
149  TVector3 beta_vector=-fCM.BoostVector();
150  Double_t gamma=fCM.Gamma();
151  Double_t beta=fCM.Beta();
152 
153  Double_t E1 = (gamma * (gamma-1.) * beta_vector.X() * beta_vector.Y() / (beta*beta)) - (beta_vector.X() * beta_vector.Y() * gamma*gamma);
154 
155  Double_t E2 = (beta_vector.Z() * gamma) - (beta_vector.Y()*(beta_vector.Y()) * gamma*gamma) + gamma + (gamma * (gamma-1.) * pow(beta_vector.Y(),2) / (beta*beta));
156 
157  Double_t E3 = (gamma * (gamma-1.) * beta_vector.Y() * beta_vector.Z() / (beta*beta)) - (beta_vector.Y() * gamma) - (beta_vector.Y() * beta_vector.Z() * gamma*gamma);
158 
159  TVector3 Polarisation(E1,E2,E3); //photon pol in CM
160  Polarisation.RotateZ(TMath::Pi()/2);
161  TLorentzVector CMGamma=fGamma;
162  //TLorentzRotation CMBoost(-fCM.BoostVector());
163  CMGamma.Boost(beta_vector);
164 
165  TVector3 zV=CMGamma.Vect().Unit();
166  TVector3 yV=Polarisation.Unit().Cross(CMGamma.Vect()).Unit();
167  TVector3 xV=yV.Cross(zV).Unit();
168 
169  TLorentzRotation CMBoost(-fCM.BoostVector());
170  TLorentzVector CMMes=CMBoost*fMes;
171  TVector3 angles(CMMes.Vect().Dot(xV),CMMes.Vect().Dot(yV),CMMes.Vect().Dot(zV));
172  fCosTh=angles.CosTheta();
