95 Bool_t RooHS1StepStudy::execute()
96 {
97  cout<<fPDFName<<endl;
98 
99  cout<<fHSRooFit<<endl;
100  fHSRooFit->RemoveDataSet();
101  fHSRooFit->CleanWSDataSets();
102 
103  //generate pseudo data set
104  fWS->loadSnapshot("initial");//load initial parameters
105 
106  RooHSEventsPDF* PDF=dynamic_cast<RooHSEventsPDF*>(fWS->pdf(fPDFName));
107  cout<<PDF<<endl;
108 
109  TTree* tree=PDF->GetGenTree();
110  Long64_t nexp=0;
111  if(fUseAll){//Just use all event in the simulated tree
112  PDF->SetUseWeightsGen(kTRUE); //must use weights for this
113  PDF->SetGeni(0);
114  nexp=tree->GetEntries();
115  }
116  else{ //Use yield parameter to decide how many events to generate
117  PDF->SetGeni(fGeni);
118  //genereate n events from poisson
119  nexp=RooRandom::randomGenerator()->Poisson(fHSRooFit->GetModel()->expectedEvents(fHSRooFit->GetVariables()));
120  }
121 
122  RooDataSet* DS=fHSRooFit->GetModel()->generate(fHSRooFit->GetVariables(),nexp);
123  tree=PDF->GetGenTree(); //this tree should have model imposed
124  Info("RooHS1StepStudy::execute() ","generated pseudo data :");
125  DS->Print();
126  PDF->initIntegrator(); //move back to reconstruced branches
127  //get generated tree, this will be flat if using weights
128  fHSRooFit->LoadDataSet(tree);
129  if(PDF->UseWeightsGen()){ //weight generated tree with model
130  //If this is not set events will just be accept or reject
131  fHSRooFit->LoadWeights(TString(fPDFName)+"Weights.root","genWeights");
132  fHSRooFit->SetWeightName(fPDFName);
133  fHSRooFit->SetDataWeightFast();
134  }
135 
136  //Fit data
137  fHSRooFit->Fit();
138 
139  //plot if requested, this takes time so generally have off
140  if(fIsPlot) fHSRooFit->PlotDataModel();
141 
142  RooFitResult* fr=(RooFitResult*)fHSRooFit->GetResult()->clone();
143  if (fr->status()==0) {
144  _ngenVar->setVal(DS->sumEntries()) ;
145  _nllVar->setVal(fr->minNll()) ;
146  storeSummaryOutput(*_params) ;
147  storeDetailedOutput(*fr) ;
148  }
149 
150  fGeni=PDF->GetGeni();
151 
152  Info("RooHS1StepStudy::execute()","Done Got to event %lld in tree",fGeni);
153  fHSRooFit->SetName(TString(GetName())+Form("_Study_%d_",summaryData()->numEntries()));
154  if(fIsPlot){fHSRooFit->SavePlots("");fHSRooFit->GetPlots()->Clear();}
155  delete DS ;
156  return kFALSE ;
157 }
